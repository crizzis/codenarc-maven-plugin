14  import javax.inject.Inject;

17  import java.util.ResourceBundle;

19  import static org.apache.maven.plugins.annotations.LifecyclePhase.SITE;
20  
21  /**
22   * Create a CodeNarc report (implies the execution of the {@code verify} goal)
23   */
24  @Getter
25  @Setter
26  @Mojo(name = "codenarc", defaultPhase = SITE)
27  @Execute(goal = "verify")
28  public class CodeNarcReportMojo extends AbstractMavenReport {
29  
30      private final CodeNarcXmlParser xmlParser;
31      private final CodeNarcReportGenerator reportGenerator;
32  
33      @Inject
34      public CodeNarcReportMojo(CodeNarcXmlParser xmlParser, CodeNarcReportGenerator reportGenerator) {
35          this.xmlParser = xmlParser;
36          this.reportGenerator = reportGenerator;
37      }
38  
39      /**
40       * Location where the generated XML report will be created
41       */
42      @Parameter(property="codenarc.xmlOutputDirectory", defaultValue = "${project.build.directory}")
43      private File xmlOutputDirectory;
44  
45      /**
46       * Set this to "true" to bypass CodeNarc report generation entirely
47       */
48      @Parameter(property="codenarc.skip", defaultValue = "false")
49      private boolean skip;
50  
51      @Override
52      public boolean canGenerateReport() {
53          File xmlOutputFile = getXmlOutputFile();
54          if (isSkip()) {
55              getLog().info("Plugin execution skipped");
56              return false;
57          }
58          if (!xmlOutputFile.exists() || !xmlOutputFile.canRead()) {
59              getLog().info(String.format("CodeNarc report %s not found - skipping HTML report generation", xmlOutputFile.getPath()));
60              return false;
61          }
62          return true;
63      }
64  
65      @Override
66      protected void executeReport(Locale locale) throws MavenReportException {
67          CodeNarcAnalysis analysis = parseAnalysis();
68          generateReport(analysis, locale);
69      }
70  
71      private void generateReport(CodeNarcAnalysis analysis, Locale locale) {
72          getLog().info("Report locale set to " + locale);
73          reportGenerator.generate(analysis, getSink(), locale);
74      }
75  
76      @Override
77      public String getOutputName() {
78          return "codenarc";
79      }
80  
81      @Override
82      public String getName(Locale locale) {
83          return getCodeNarcMessages(locale).getString("report.codenarc.name");
84      }
85  
86      @Override
87      public String getDescription(Locale locale) {
88          return getCodeNarcMessages(locale).getString("report.codenarc.description");
89      }
90  
91      private ResourceBundle getCodeNarcMessages(Locale locale) {
92          return ResourceBundle.getBundle("codenarc-messages", locale);
93      }
94  
95      private CodeNarcAnalysis parseAnalysis() throws MavenReportException {
96          File outputFile = getXmlOutputFile();
97          try {
98              getLog().info("CodeNarc report XML found, parsing");
99              CodeNarcAnalysis analysis = xmlParser.parse(outputFile);
100             getLog().info("Parsing completed");
101             return analysis;
102         } catch (CodeNarcXmlParser.XmlParserException e) {
103             throw new MavenReportException(String.format("Could not parse: %s", outputFile.getAbsolutePath()), e);
104         }
105     }
106 
107     private File getXmlOutputFile() {
108         return new File(getXmlOutputDirectory(), "CodeNarc.xml");
109     }
110 }
