20  /**
21   * CodeNarc report generator. Takes an instance of {@link CodeNarcAnalysis} as input
22   */
23  @Named
24  @Singleton
25  public class CodeNarcReportGenerator implements Localizable {
26  
27      private static final int MAX_PRIORITY = 3;
28      private static final String KEY_VALUE_SEPARATOR = ": ";
29  
30      private final ResultWalkerultWalker.html#ResultWalker">ResultWalker resultWalker = new ResultWalker();
31  
32      public void generate(CodeNarcAnalysis input, Sink sink, Locale locale) {
33          sink.head();
34          generateHead(input, sink, locale);
35          sink.head_();
36          sink.body();
37          generateBody(input, sink, locale);
38          sink.body_();
39      }
40  
41      private void generateHead(CodeNarcAnalysis input, Sink sink, Locale locale) {
42          sink.title();
43          sink.text(getCodeNarcMessages(locale).getString("report.codenarc.name"));
44          sink.title_();
45      }
46  
47      private void generateBody(CodeNarcAnalysis input, Sink sink, Locale locale) {
48          generateHeading(input, sink, locale);
49          generateSummary(input, sink, locale);
50          if (hasViolations(input)) {
51              generatePackageSummary(input, sink, locale);
52              generateFileViolations(input, sink, locale);
53          }
54      }
55  
56      private void generateHeading(CodeNarcAnalysis input, Sink sink, Locale locale) {
57          sink.section1();
58          printSectionTitle(sink, getCodeNarcMessages(locale).getString("report.codenarc.name"));
59          printParagraph(sink, getCodeNarcMessages(locale).getString("report.codenarc.content_description"));
60          printKeyValue(sink, getCodeNarcMessages(locale).getString("report.codenarc.version_caption"), input.getCodeNarcVersion());
61          printKeyValue(sink, getCodeNarcMessages(locale).getString("report.codenarc.generation_time_caption"), input.getReportTimestamp());
62          sink.section1_();
63      }
64  
65      private void generateSummary(CodeNarcAnalysis input, Sink sink, Locale locale) {
66          sink.section1();
67          printSectionTitle(sink, getCodeNarcMessages(locale).getString("report.codenarc.summary"));
68          printTable(sink, new SummaryTableRenderer(locale), List.of((DirectoryResults) input.getResults()));
69          sink.section1_();
70      }
71  
72      private void generatePackageSummary(CodeNarcAnalysis input, Sink sink, Locale locale) {
73          sink.section1();
74          printSectionTitle(sink, getCodeNarcMessages(locale).getString("report.codenarc.package_summary"));
75          Iterator<String> sourceRootDirectories = input.getSourceDirectories().iterator();
76          input.getResults().getChildren().forEach(sourceRoot -> {
77              if (sourceRootDirectories.hasNext()) {
78                  printSourceDirectoryTitle(sink, sourceRootDirectories.next(), locale);
79              }
80              printResultTable(sink, new PackageSummaryTableRenderer(locale), (Results) sourceRoot, DIRECTORIES_WITH_FILES);
81          });
82          sink.section1_();
83      }
84  
85      private void generateFileViolations(CodeNarcAnalysis input, Sink sink, Locale locale) {
86          sink.section1();
87          printSectionTitle(sink, getCodeNarcMessages(locale).getString("report.codenarc.files"));
88          final Iterator<String> sourceRootDirectories = input.getSourceDirectories().iterator();
89          final CurrentPackageContext context = new CurrentPackageContext();
90          resultWalker.walk(input.getResults(), DIRECTORIES_WITH_FILES.or(FILES).or(SOURCE_ROOTS), results -> {
91              if (results.isFile()) {
92                  printSubSubSectionTitle(sink, toFilePath(context.getCurrentPackage(), results));
93                  printTable(sink, new ViolationTableRenderer(locale), results.getViolations());
94              } else {
95                  if (isSourceRoot(results) && sourceRootDirectories.hasNext()) {
96                      printSourceDirectoryTitle(sink, sourceRootDirectories.next(), locale);
97                  }
98                  context.setCurrentPackage(results);
99              }
100         });
101         sink.section1_();
102     }
103 
104     private void printSectionTitle(Sink sink, String title) {
105         sink.sectionTitle1();
106         sink.text(title);
107         sink.sectionTitle1_();
108     }
109 
110     private void printSourceDirectoryTitle(Sink sink, String sourceDirectory, Locale locale) {
111         sink.sectionTitle2();
112         sink.text(getCodeNarcMessages(locale).getString("report.codenarc.source_directory") + KEY_VALUE_SEPARATOR);
113         sink.italic();
114         sink.text(sourceDirectory);
115         sink.italic_();
116         sink.sectionTitle2_();
117     }
118 
119     private void printSubSubSectionTitle(Sink sink, String title) {
120         sink.sectionTitle3();
121         sink.text(title);
122         sink.sectionTitle3_();
123     }
124 
125     private void printKeyValue(Sink sink, String key, String value) {
126         sink.paragraph();
127         sink.text(key + KEY_VALUE_SEPARATOR);
128         sink.italic();
129         sink.text(value);
130         sink.italic_();
131         sink.paragraph_();
132     }
133 
134     private void printParagraph(Sink sink, String paragraphText) {
135         sink.paragraph();
136         sink.text(paragraphText);
137         sink.paragraph_();
138     }
139 
140     private <T> void printTable(Sink sink, TableRenderer<T> renderer, List<T> elements) {
141         renderer.renderTable(sink, elements);
142     }
143 
144     private void printResultTable(Sink sink, TableRenderer<Results> renderer, Results root, Predicate<Results> include) {
145         renderer.renderTable(sink, renderingCallback -> resultWalker.walk(root, include, renderingCallback));
146     }
147 
148     private boolean hasViolations(CodeNarcAnalysis input) {
149         return input.getResults().getNumberOfFilesWithViolations(MAX_PRIORITY, true) > 0;
150     }
151 
152     private String toFilePath(Results directory, Results file) {
153         return (StringUtils.isBlank(directory.getPath()) ? "" : directory.getPath() + "/") + file.getPath();
154     }
155 
156     private boolean isSourceRoot(Results results) {
157         return results.getPath().isBlank();
158     }
