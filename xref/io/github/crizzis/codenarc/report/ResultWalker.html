5   import java.util.ArrayDeque;
6   import java.util.Comparator;
7   import java.util.Deque;
8   import java.util.Objects;
9   import java.util.function.Consumer;
10  import java.util.function.Predicate;
11  
12  /**
13   * Performs a DFS traversal of CodeNarc results
14   */
15  @SuppressWarnings("unchecked")
16  class ResultWalker {
17  
18      static final Predicate<Results> FILES = Results::isFile;
19      static final Predicate<Results> DIRECTORIES = Predicate.not(FILES);
20      static final Predicate<Results> SOURCE_ROOTS = DIRECTORIES.and(results -> Objects.nonNull(results.getPath()) && results.getPath().isBlank());
21      static final Predicate<Results> DIRECTORIES_WITH_FILES = results -> results.getChildren().stream().anyMatch(FILES);
22  
23      private final Deque<Results> toVisit = new ArrayDeque<>();
24  
25      void walk(Results results, Predicate<Results> filter, Consumer<Results> action) {
26          if (filter.test(results)) {
27              action.accept(results);
28          }
29          results.getChildren().stream()
30                  .sorted(Comparator.comparing(Results::isFile).reversed())
31                  .forEachOrdered(child -> walk((Results) child, filter, action));
32      }
33  }
