1   package io.github.crizzis.codenarc.report;
2   
3   import lombok.Getter;
4   import org.apache.maven.doxia.sink.Sink;
5   import org.codenarc.results.DirectoryResults;
6   
7   import java.util.List;
8   import java.util.Locale;
9   import java.util.ResourceBundle;
10  
11  class SummaryTableRenderer implements TableRenderer<DirectoryResults>, Localizable {
12  
13      private static final int PRIORITY_ONE = 1;
14      private static final int PRIORITY_TWO = 2;
15      private static final int PRIORITY_THREE = 3;
16  
17      @Getter
18      private final List<String> headers;
19  
20      SummaryTableRenderer(Locale locale) {
21          ResourceBundle messages = getCodeNarcMessages(locale);
22          this.headers = List.of(
23                  messages.getString("report.codenarc.total_files"),
24                  messages.getString("report.codenarc.files_with_violations"),
25                  messages.getString("report.codenarc.total_violations"),
26                  messages.getString("report.codenarc.priority_one_violations"),
27                  messages.getString("report.codenarc.priority_two_violations"),
28                  messages.getString("report.codenarc.priority_three_violations"));
29      }
30  
31      @Override
32      public void renderCell(Sink sink, DirectoryResults element, int index) {
33          sink.text(String.valueOf(getValue(element, index)));
34      }
35  
36      public int getValue(DirectoryResults element, int index) {
37          switch (index) {
38              case 0:
39                  return element.getTotalNumberOfFiles(false);
40              case 1:
41                  return element.getNumberOfFilesWithViolations(PRIORITY_THREE);
42              case 2:
43                  return (element.getNumberOfViolationsWithPriority(PRIORITY_ONE)
44                          + element.getNumberOfViolationsWithPriority(PRIORITY_TWO)
45                          + element.getNumberOfViolationsWithPriority(PRIORITY_THREE));
46              case 3:
47                  return element.getNumberOfViolationsWithPriority(PRIORITY_ONE);
48              case 4:
49                  return element.getNumberOfViolationsWithPriority(PRIORITY_TWO);
50              case 5:
51                  return element.getNumberOfViolationsWithPriority(PRIORITY_THREE);
52              default:
53                  throw new IllegalArgumentException("Invalid table cell index");
54          }
55      }
56  }
