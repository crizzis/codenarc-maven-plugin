1   package io.github.crizzis.codenarc.report;
2   
3   import org.apache.maven.doxia.sink.Sink;
4   
5   import java.util.List;
6   import java.util.function.Consumer;
7   
8   public interface TableRenderer<T> {
9   
10      default void renderTable(Sink sink, Consumer<Consumer<T>> callback) {
11          sink.table();
12          renderHeader(sink);
13          callback.accept(element -> renderRow(sink, element));
14          sink.table_();
15      }
16  
17      default void renderTable(Sink sink, Iterable<T> elements) {
18          renderTable(sink, elements::forEach);
19      }
20  
21      default void renderHeader(Sink sink) {
22          sink.tableRow();
23          getHeaders().forEach(header -> {
24              sink.tableHeaderCell();
25              sink.text(header);
26              sink.tableHeaderCell_();
27          });
28          sink.tableRow_();
29      }
30  
31      List<String> getHeaders();
32  
33      default void renderRow(Sink sink, T element) {
34          sink.tableRow();
35          for (int i = 0; i < getColumnCount(); ++i) {
36              sink.tableCell();
37              renderCell(sink, element, i);
38              sink.tableCell_();
39          }
40          sink.tableRow_();
41      }
42  
43      void renderCell(Sink sink, T element, int index);
44  
45      default int getColumnCount() {
46          return getHeaders().size();
47      }
48  }
