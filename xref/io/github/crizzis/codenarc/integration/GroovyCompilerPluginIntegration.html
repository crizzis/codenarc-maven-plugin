9   /**
10   * Defines an entry point for integration with Groovy compiler plugins.
11   *
12   * Implementations of this interface should scan the project build configuration
13   * in search of a specific Groovy compiler plugin and, if found, extract compile sources configuration
14   * from the configuration of that plugin. The obtained compile sources configuration, if found, will then be used
15   * if no custom {@code sources}/{@code testSources} configuration has been specified for the {@code verify} goal
16   */
17  public interface GroovyCompilerPluginIntegration {
18  
19      Optional<FileSet[]> getSources(Build build, MavenSession mavenSession);
20  
21      Optional<FileSet[]> getTestSources(Build build, MavenSession mavenSession);
22  }
