28  /**
29   *  CodeNarc XML report parser. This class converts XML CodeNarc output files into
30   *  instances of {@link org.codenarc.results.Results}
31   */
32  @Named
33  @Singleton
34  public class CodeNarcXmlParser {
35  
36      private static final XMLInputFactory FACTORY = XMLInputFactory.newInstance();
37  
38      public CodeNarcAnalysis parse(File xmlReport) throws XmlParserException {
39          try (InputStream xmlInput = new FileInputStream(xmlReport)) {
40              XMLEventReader xmlEventReader = FACTORY.createFilteredReader(FACTORY.createXMLEventReader(xmlInput),
41                      NodesAttributesAndCharactersOnly.filter());
42              CodeNarcXmlEventProcessor processor = new CodeNarcXmlEventProcessor(xmlEventReader);
43              processor.process();
44              return processor.getAnalysis();
45          } catch (IOException | XMLStreamException | ClassCastException
46                  | IllegalArgumentException | IllegalStateException | NoSuchElementException e) {
47              throw new XmlParserException(e);
48          }
49      }
50  
51      @RequiredArgsConstructor
52      @Getter
53      private static class CodeNarcXmlEventProcessor {
54  
55          private final XMLEventReader eventReader;
56  
57          private final CodeNarcXmlEventConsumerEventConsumer.html#CodeNarcXmlEventConsumer">CodeNarcXmlEventConsumer consumer = new CodeNarcXmlEventConsumer();
58  
59          private final Map<String, Consumer<EndElement>> END_ELEMENT_CONSUMER_METHODS = Map.ofEntries(
60                  entry("CodeNarc", consumer::consumeEndCodeNarc),
61                  entry("Report", consumer::consumeEndReport),
62                  entry("Project", consumer::consumeEndProject),
63                  entry("SourceDirectory", consumer::consumeEndSourceDirectory),
64                  entry("PackageSummary", consumer::consumeEndPackageSummary),
65                  entry("Package", consumer::consumeEndPackage),
66                  entry("File", consumer::consumeEndFile),
67                  entry("Violation", consumer::consumeEndViolation),
68                  entry("SourceLine", consumer::consumeEndSourceLine),
69                  entry("Message", consumer::consumeEndMessage)
70          );
71  
72          private final Map<String, Consumer<StartElement>> START_ELEMENT_CONSUMER_METHODS = Map.ofEntries(
73                  entry("CodeNarc", consumer::consumeStartCodeNarc),
74                  entry("Report", consumer::consumeStartReport),
75                  entry("Project", consumer::consumeStartProject),
76                  entry("SourceDirectory", consumer::consumeStartSourceDirectory),
77                  entry("PackageSummary", consumer::consumeStartPackageSummary),
78                  entry("Package", consumer::consumeStartPackage),
79                  entry("File", consumer::consumeStartFile),
80                  entry("Violation", consumer::consumeStartViolation),
81                  entry("SourceLine", consumer::consumeStartSourceLine),
82                  entry("Message", consumer::consumeStartMessage)
83          );
84  
85          void process() throws XMLStreamException {
86              while (eventReader.hasNext()) {
87                  XMLEvent xmlEvent = eventReader.nextEvent();
88                  if (xmlEvent.isStartElement()) {
89                      handleStartElement(eventReader, xmlEvent.asStartElement());
90                  } else if (xmlEvent.isEndElement()) {
91                      handleEndElement(xmlEvent.asEndElement());
92                  } else if (xmlEvent.isCharacters()) {
93                      handleCharacters(xmlEvent.asCharacters());
94                  }
95              }
96          }
97  
98          private void handleCharacters(Characters characters) {
99              consumer.consumeCharacters(characters);
100         }
101 
102         private void handleEndElement(EndElement endElement) {
103             String name = endElement.getName().getLocalPart();
104             END_ELEMENT_CONSUMER_METHODS.getOrDefault(name, element -> verifyIgnoredTag(name)).accept(endElement);
105         }
106 
107         private void handleStartElement(XMLEventReader eventReader, StartElement startElement) throws XMLStreamException {
108             String name = startElement.getName().getLocalPart();
109             if (consumer.getSkippedTags().contains(name)) {
110                 skipUntilEnd(eventReader, name);
111             }
112             START_ELEMENT_CONSUMER_METHODS.getOrDefault(name, element -> verifyIgnoredTag(name)).accept(startElement);
113         }
114 
115         private void skipUntilEnd(XMLEventReader eventReader, String name) throws XMLStreamException {
116             XMLEvent xmlEvent;
117             do {
118                 xmlEvent = eventReader.nextEvent();
119             } while (!xmlEvent.isEndElement() || !xmlEvent.asEndElement().getName().getLocalPart().equals(name));
120         }
121 
122         private void verifyIgnoredTag(String name) {
123             if (!consumer.getIgnoredTags().contains(name)) {
124                 throw new IllegalArgumentException("Unrecognized tag " + name);
125             }
126         }
127 
128         public CodeNarcAnalysis getAnalysis() {
129             return consumer.getAnalysis();
130         }
131     }
132 
133     private static class NodesAttributesAndCharactersOnly implements EventFilter {
134 
135         private static final List<Integer> IGNORED_EVENTS = List.of(START_DOCUMENT, END_DOCUMENT, ENTITY_DECLARATION,
136                 ENTITY_REFERENCE, COMMENT, PROCESSING_INSTRUCTION, NAMESPACE, SPACE, NOTATION_DECLARATION);
137 
138         public static EventFilter filter() {
139             return new NodesAttributesAndCharactersOnly();
140         }
141 
142         @Override
143         public boolean accept(XMLEvent event) {
144             if (event.isCharacters()) {
145                 return !event.asCharacters().isWhiteSpace();
146             }
147             return !IGNORED_EVENTS.contains(event.getEventType());
148         }
149     }
