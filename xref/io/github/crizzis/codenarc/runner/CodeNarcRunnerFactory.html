1   package io.github.crizzis.codenarc.runner;
2   
3   import org.codenarc.CodeNarcRunner;
4   import org.codenarc.report.XmlReportWriter;
5   
6   import javax.inject.Named;
7   import javax.inject.Singleton;
8   import java.util.List;
9   
10  /**
11   * Configures a {@link org.codenarc.CodeNarcRunner} used for running the {@code verify} goal
12   */
13  @Named
14  @Singleton
15  public class CodeNarcRunnerFactory {
16  
17      public CodeNarcRunner newCodeNarcRunner(CodeNarcConfig config) {
18          CodeNarcRunner runner = new CodeNarcRunner();
19          runner.setRuleSetFiles(joinedRuleSets(config));
20          if (config.isGenerateXmlReport()) {
21              configureXmlOutput(config, runner);
22          }
23          runner.setSourceAnalyzer(new FileSetSourceAnalyzer(config.getFileSets()));
24          return runner;
25      }
26  
27      private void configureXmlOutput(CodeNarcConfig config, CodeNarcRunner runner) {
28          XmlReportWriter xmlReportWriter = new XmlReportWriter();
29          xmlReportWriter.setTitle(config.getProjectName());
30          xmlReportWriter.setOutputFile(config.getOutputFile().getAbsolutePath());
31          runner.setReportWriters(List.of(xmlReportWriter));
32      }
33  
34      private String joinedRuleSets(CodeNarcConfig config) {
35          return String.join(",", config.getRuleSets());
36      }
37  
38  }
