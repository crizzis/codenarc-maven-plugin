1   package io.github.crizzis.codenarc.runner;
2   
3   import lombok.NonNull;
4   import lombok.RequiredArgsConstructor;
5   import org.apache.maven.model.FileSet;
6   import org.codenarc.analyzer.AbstractSourceAnalyzer;
7   import org.codenarc.analyzer.FilesystemSourceAnalyzer;
8   import org.codenarc.results.DirectoryResults;
9   import org.codenarc.results.Results;
10  import org.codenarc.ruleset.RuleSet;
11  
12  import java.util.List;
13  import java.util.stream.Collector;
14  import java.util.stream.Collectors;
15  
16  /**
17   * A composite-style analyzer that supports multiple {@link FileSet}s
18   * by using multiple {@link FilesystemSourceAnalyzer}s
19   */
20  @RequiredArgsConstructor
21  class FileSetSourceAnalyzer extends AbstractSourceAnalyzer {
22  
23      @NonNull
24      private final List<FileSet> fileSets;
25  
26      @Override
27      @SuppressWarnings("unchecked")
28      public Results analyze(RuleSet ruleSet) {
29          return fileSets.stream()
30                  .map(fileSet -> analyze(fileSet, ruleSet))
31                  .collect(Collector.of(DirectoryResults::new, DirectoryResults::addChild, (left, right) -> {
32                      ((List<Results>) right.getChildren()).forEach(left::addChild);
33                      return left;
34                  }));
35      }
36  
37      private Results analyze(FileSet fileSet, RuleSet ruleSet) {
38          FilesystemSourceAnalyzer analyzer = new FilesystemSourceAnalyzer();
39          analyzer.setBaseDirectory(fileSet.getDirectory());
40          analyzer.setIncludes(String.join(",", fileSet.getIncludes()));
41          analyzer.setExcludes(String.join(",", fileSet.getExcludes()));
42          return (Results) analyzer.analyze(ruleSet).getChildren().get(0);
43      }
44  
45      @Override
46      public List<String> getSourceDirectories() {
47          return fileSets.stream()
48                  .map(FileSet::getDirectory)
49                  .collect(Collectors.toList());
50      }
51  }
