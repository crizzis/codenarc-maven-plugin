35      @MavenProjectTest("/projects/verify-minimal-config-no-violations")
36      void verifyMinimalConfig_shouldSucceed(Verifier verifier, @ProjectRoot File projectRoot) throws Exception {
37          //when
38          verifier.executeGoal("verify");
39  
40          //then
41          verifier.assertFilePresent("target/CodeNarc.xml");
42          Document codeNarcReport = parseXml(projectRoot, "target/CodeNarc.xml");
43          assertThat(codeNarcReport, hasProjectName("verify-minimal-config-no-violations"));
44          verifier.verifyTextInLog("CodeNarc completed: (p1=0; p2=0; p3=0)");
45          verifier.verifyErrorFreeLog();
46      }
47  
48      @MavenProjectTest("/projects/verify-minimal-config-violations-above-threshold")
49      void verifyMinimalConfig_shouldFail_whenViolationsAboveThreshold(Verifier verifier, @ProjectRoot File projectRoot)
50              throws Exception {
51          //when, then
52          VerificationException exception = assertThrows(VerificationException.class, () -> verifier.executeGoal("verify"));
53          assertThat(exception.getMessage(), startsWith("Exit code was non-zero"));
54          verifier.verifyTextInLog("CodeNarc completed: (p1=0; p2=5; p3=0)");
55          verifier.verifyTextInLog("totalPriority2Violations exceeded threshold of 3 errors with 5");
56  
57          verifier.assertFilePresent("target/CodeNarc.xml");
58          Document codeNarcReport = parseXml(projectRoot, "target/CodeNarc.xml");
59          assertThat(codeNarcReport, includesSourceDirectory(1, "src/main/groovy"));
60      }
61  
62      @MavenProjectTest("/projects/verify-custom-config")
63      void verifyCustomConfig_shouldRespectCustomSettings(Verifier verifier, @ProjectRoot File projectRoot)
64              throws Exception {
65          //when
66          verifier.executeGoal("codenarc:verify");
67  
68          //then
69          verifier.verifyTextInLog("Using rule sets: rulesets/formatting.xml");
70          verifier.verifyTextInLog("custom.xml");
71          verifier.verifyTextInLog("CodeNarc completed: (p1=0; p2=6; p3=10)");
72  
73          verifier.assertFilePresent("target/reports/CodeNarc.xml");
74          Document codeNarcReport = parseXml(projectRoot, "target/reports/CodeNarc.xml");
75          assertThat(codeNarcReport, includesSourceDirectory(1, "src/main/groovy"));
76          assertThat(codeNarcReport, includesSourceDirectory(2, "src/main/additional"));
77          assertThat(codeNarcReport, includesSourceDirectory(3, "src/test/groovy"));
78          assertThat(codeNarcReport, includesFile("TestClassIncluded.groovy"));
79          assertThat(codeNarcReport, not(includesFile("TestClassWithSomeMoreViolationsExcluded.groovy")));
80      }

83      void verifyCustomConfig_shouldNotRun_whenExecutionSkipped(Verifier verifier) throws Exception {

96      @MavenProjectTest("/projects/verify-preexisting-report")
97      void verifyPreexistingReport_shouldConsumePreexistingReportCorrectly(Verifier verifier, @ProjectRoot File projectRoot)
98              throws Exception {
99          //when
100         verifier.executeGoal("codenarc:verify@generate-report");
101 
102         //then
103         verifier.verifyTextInLog("CodeNarc completed: (p1=0; p2=5; p3=0)");
104         verifier.assertFilePresent("target/CodeNarc.xml");
105 
106         //when, then
107         verifier.resetStreams();
108         verifier.setAutoclean(false);
109         assertThrows(VerificationException.class, () -> verifier.executeGoal("codenarc:verify@consume-report"));
110         verifier.verifyTextInLog("Parsing completed: (p1=0; p2=5; p3=0)");
111         verifier.verifyTextInLog("totalPriority2Violations exceeded threshold of 3 errors with 5");
112     }

114     private Matcher<Node> includesSourceDirectory(int index, String sourceDirectory) {
115         return hasXPath("/CodeNarc/Project/SourceDirectory[" + index + "]", equalTo(sourceDirectory));
116     }
117 
118     private Matcher<Node> hasProjectName(String projectName) {
119         return hasXPath("/CodeNarc/Project/@title", equalTo(projectName));
120     }
121 
122     private Matcher<Node> includesFile(String filename) {
123         return hasXPath("/CodeNarc/Package/File[@name='" + filename + "']");
124     }
125 
126     private Document parseXml(File parent, String path) throws SAXException, IOException {
127         return xmlBuilder.parse(new File(parent, path));
128     }
