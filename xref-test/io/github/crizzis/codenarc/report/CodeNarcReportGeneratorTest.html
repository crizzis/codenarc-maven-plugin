57      @ParameterizedTest(name = "should generate correct report for {2}")
58      @MethodSource
59      void shouldGenerateCorrectReport_whenValidAnalysis(CodeNarcAnalysis input, String expected, String description) {
60          //when
61          generator.generate(input, sinkMock.initialize(), ENGLISH);
62          String actual = sinkMock.terminate();
63  
64          //then
65          Diff reportDiff = DiffBuilder.compare(expected)
66                  .withTest(Input.fromString(actual))
67                  .ignoreWhitespace()
68                  .checkForSimilar()
69                  .build();
70          assertThat(reportDiff.toString(), !reportDiff.hasDifferences());

167     private static CodeNarcAnalysis analysis(String title,
168                                       String version,
169                                       String timestamp,
170                                       List<String> sourceDirectories,
171                                       Results results) {
172         CodeNarcAnalysis analysis = new CodeNarcAnalysis();
173         analysis.setProjectTitle(title);
174         analysis.setCodeNarcVersion(version);
175         analysis.setReportTimestamp(timestamp);
176         analysis.setSourceDirectories(sourceDirectories);
177         analysis.setResults(results);
178         return analysis;
179     }
