35      @MavenProjectTest("/projects/codenarc-minimal-config")
36      void codenarcMinimalConfig_shouldSucceed(Verifier verifier, @ProjectRoot File projectRoot) throws Exception {
37          //when
38          verifier.executeGoal("site");
39  
40          //then
41          verifier.verifyErrorFreeLog();
42          verifier.assertFilePresent("target/CodeNarc.xml");
43          verifier.assertFilePresent("target/site/codenarc.html");
44          Document codenarcHtml = parseXml(projectRoot, "target/site/codenarc.html");
45          assertThat(codenarcHtml, hasSection(1, "CodeNarc Report"));
46          assertThat(codenarcHtml, hasSection(2, "Summary"));
47          assertThat(codenarcHtml, hasSection(3, "Package Summary"));
48          assertThat(codenarcHtml, hasSection(4, "Files"));
49          assertThat(codenarcHtml, hasFileViolation("com/example/TestClassWithSomeViolations.groovy", "BrokenNullCheck"));
50          assertThat(codenarcHtml, hasFileViolation("com/example/TestClassWithSomeViolations.groovy", "DuplicateMapKey"));
51          assertThat(codenarcHtml, hasFileViolation("com/example/nestedpackage/TestClassWithSomeMoreViolations.groovy", "DeadCode"));
52          assertThat(codenarcHtml, hasFileViolation("com/example/nestedpackage/TestClassWithSomeMoreViolations.groovy", "EmptyIfStatement"));
53      }
54  
55      @MavenProjectTest("/projects/codenarc-custom-config")
56      void codenarcCustomConfig_shouldSucceed(Verifier verifier, @ProjectRoot File projectRoot) throws Exception {
57          //when
58          verifier.executeGoal("codenarc:codenarc");
59  
60          //then
61          verifier.verifyErrorFreeLog();
62          verifier.assertFilePresent("target/reports/CodeNarc.xml");
63          verifier.assertFilePresent("target/site/codenarc.html");
64          Document codenarcHtml = parseXml(projectRoot, "target/site/codenarc.html");
65          assertThat(codenarcHtml, hasSourceDirectory("src/main/additional"));
66          assertThat(codenarcHtml, hasFileViolation("com/example/TestClassIncluded.groovy", "ClassEndsWithBlankLine"));
67          assertThat(codenarcHtml, hasFileViolation("com/example/TestClassWithSomeViolationsIncluded.groovy", "SpaceAroundMapEntryColon"));
68          assertThat(codenarcHtml, not(hasFileViolation("com/example/nestedpackage/TestClassWithSomeMoreViolationsExcluded.groovy", "DeadCode")));
69      }
70  
71      private Matcher<Node> hasSourceDirectory(String sourceDirectoryPath) {
72          return hasXPath("//h3[text() = 'Source Directory: ']/i[text() = '" + sourceDirectoryPath + "']");
73      }
74  
75      private Matcher<Node> hasSection(int sectionIndex, String caption) {
76          return hasXPath("//div[@id='contentBox']/*[@class='section'][" + sectionIndex + "]/h2/text()", equalTo(caption));
77      }
78  
79      private Matcher<Node> hasFileViolation(final String fileName, final String violation) {
80          return hasXPath("//h4[text() = '" + fileName + "']/following-sibling::table[1]/tr/td[text() = '" + violation + "']");
81      }
82  
83      private Document parseXml(File parent, String path) throws SAXException, IOException {
84          return xmlBuilder.parse(new File(parent, path));
85      }
